Here are the architectural instructions to build a Hybrid Lending Platform.

This system allows you to define your own strict underwriting rules (Internal) while simultaneously matching borrowers against a database of external lender criteria (Marketplace).

Phase 1: The "Dynamic Rules" Database
We stop hard-coding rules. Instead, we store "Lending Programs" in the database. A program can be marked as INTERNAL (You) or EXTERNAL (Partner).

shared/schema.ts

TypeScript

import { pgTable, serial, text, integer, decimal, boolean, jsonb } from "drizzle-orm/pg-core";

export const lenders = pgTable("lenders", {
  id: serial("id").primaryKey(),
  name: text("name").notNull(),
  type: text("type").default("EXTERNAL"), // 'INTERNAL' or 'EXTERNAL'
  logoUrl: text("logo_url"),
});

export const loanPrograms = pgTable("loan_programs", {
  id: serial("id").primaryKey(),
  lenderId: integer("lender_id").references(() => lenders.id),
  name: text("name").notNull(), // e.g., "Bistelligent Direct 30yr" or "Chase Jumbo"
  
  // 1. THE CREDIT BOX (Knockout Rules)
  minCreditScore: integer("min_credit_score").default(620),
  maxDTI: integer("max_dti").default(45), // Debt-to-Income cap
  maxLTV: integer("max_ltv").default(95), // Loan-to-Value cap
  minLoanAmount: integer("min_loan_amount").default(50000),
  maxLoanAmount: integer("max_loan_amount").default(766550),

  // 2. THE PRICING ENGINE
  baseRate: decimal("base_rate").notNull(), // Current starting rate (e.g. 6.5)
  
  // 3. LOAN LEVEL PRICE ADJUSTMENTS (The Logic)
  // Stores array of rules: { field: "creditScore", operator: "<", value: 680, adjustment: 0.25 }
  adjustments: jsonb("adjustments").$type<PricingRule[]>(),
  
  isActive: boolean("is_active").default(true),
});

// TypeScript type for the JSONB column
type PricingRule = {
  field: "creditScore" | "ltv" | "propertyType" | "loanPurpose";
  operator: "<" | ">" | "==";
  value: string | number;
  adjustment: number; // e.g., +0.125 or -0.25
};
Phase 2: The Universal Logic Engine
This engine doesn't care if the lender is you or a partner. It takes a Borrower and a Program, and checks if they fit.

server/engine/UniversalUnderwriter.ts

TypeScript

interface MatchResult {
  isEligible: boolean;
  finalRate: number;
  disqualificationReason?: string;
}

export class UniversalUnderwriter {
  
  public evaluate(borrower: any, program: any): MatchResult {
    const { creditScore, dti, ltv, loanAmount, propertyType } = borrower;

    // 1. KNOCKOUT RULES (The "No" Gate)
    if (creditScore < program.minCreditScore) 
      return { isEligible: false, finalRate: 0, disqualificationReason: "Credit Score too low" };
    
    if (dti > program.maxDTI) 
      return { isEligible: false, finalRate: 0, disqualificationReason: "DTI too high" };

    if (ltv > program.maxLTV) 
      return { isEligible: false, finalRate: 0, disqualificationReason: "Insufficient Equity/Down Payment" };
    
    if (loanAmount < program.minLoanAmount || loanAmount > program.maxLoanAmount)
      return { isEligible: false, finalRate: 0, disqualificationReason: "Loan amount out of bounds" };


    // 2. PRICING ENGINE (The Rate Calculator)
    let rate = parseFloat(program.baseRate);
    const rules = program.adjustments as any[];

    if (rules) {
      for (const rule of rules) {
        let match = false;
        
        // Evaluate dynamic rules stored in DB
        if (rule.field === "creditScore") {
           if (rule.operator === "<" && creditScore < rule.value) match = true;
           if (rule.operator === ">" && creditScore > rule.value) match = true;
        }
        else if (rule.field === "propertyType") {
           if (rule.value === propertyType) match = true;
        }
        // ... add logic for LTV, etc.

        if (match) rate += rule.adjustment;
      }
    }

    return { isEligible: true, finalRate: parseFloat(rate.toFixed(3)) };
  }
}
Phase 3: The "Traffic Controller" (Router)
This API route fetches all active programs, runs the engine on them simultaneously, and returns the unified list.

server/routes.ts (The Matching Endpoint)

TypeScript

import { UniversalUnderwriter } from "./engine/UniversalUnderwriter";
import { db } from "./db";
import { loanPrograms, lenders } from "@shared/schema";
import { eq } from "drizzle-orm";

app.post("/api/get-offers", async (req, res) => {
  const borrower = req.body; // { creditScore: 720, dti: 35, ... }
  const engine = new UniversalUnderwriter();
  const offers = [];

  // 1. Fetch ALL Active Programs (Internal & External)
  const allPrograms = await db.select()
    .from(loanPrograms)
    .innerJoin(lenders, eq(loanPrograms.lenderId, lenders.id))
    .where(eq(loanPrograms.isActive, true));

  // 2. Run the Engine on every program
  for (const row of allPrograms) {
    const program = row.loan_programs;
    const lender = row.lenders;

    const result = engine.evaluate(borrower, program);

    if (result.isEligible) {
      offers.push({
        lenderName: lender.name,
        lenderType: lender.type, // "INTERNAL" or "EXTERNAL"
        programName: program.name,
        rate: result.finalRate,
        monthlyPayment: calculateMortgage(borrower.loanAmount, result.finalRate),
        logo: lender.logoUrl
      });
    }
  }

  // 3. Sort: Lowest Rate First, but prioritize Internal if rates are close
  offers.sort((a, b) => a.rate - b.rate);

  res.json(offers);
});

How to execute this plan:
Set up the Database first: Run the schema code (Phase 1) in Drizzle/Postgres. This creates the container for your logic.

Seed your data: Create one lender called "Bistelligent Direct" (Internal) and one called "Rocket Mortgage" (External). Give them different baseRate and minCreditScore values in the database.

Implement the Logic: Copy the UniversalUnderwriter class (Phase 2).

Connect the Frontend: When your "Smart Form" submits, point it to the /api/get-offers endpoint (Phase 3).

Visualize: Use the "Offer Dashboard" UI component from the previous prompt to display the array of offers returned by the API.
